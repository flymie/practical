使用方式：
<pre>
new lineCanvas({
    el: document.getElementById("canvas"),//绘制canvas的父级div
    clearEl: document.getElementById("clearCanvas"),//清除按钮
    saveEl: document.getElementById("saveCanvas"),//保存按钮
    linewidth:1,//线条粗细，选填
    color:"#000",//线条颜色，选填
    background:"#fff"//线条背景，选填
});
</pre>
